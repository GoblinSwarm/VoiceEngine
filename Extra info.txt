# 🔄 Pipeline Layer

## VoiceInput

```cpp
class VoiceInput {
private:
    AudioInput& audioInput;
    AudioPreprocessor& preprocessor;
    ISTTEngine& stt;

public:
    std::string captureAndTranscribe() {
        auto raw = audioInput.readChunk();
        auto processed = preprocessor.normalize(raw);
        return stt.transcribe(processed);
    }
};
```

---

## VoiceOutput

```cpp
class VoiceOutput {
private:
    AudioOutput& audioOutput;
    ITTSEngine& tts;

public:
    void synthesizeAndPlay(const std::string& text) {
        auto audio = tts.synthesize(text);
        audioOutput.play(audio);
    }
};
```

---

# 🧠 Orchestration Layer

## VoiceOrchestrator

```cpp
class VoiceOrchestrator {
private:
    VoiceInput& input;
    VoiceOutput& output;
    CommandRouter& router;

public:
    void runOnce() {
        std::string text = input.captureAndTranscribe();
        std::string response = router.handle(text);
        output.synthesizeAndPlay(response);
    }
};
```

# 🧠 CommandRouter (clave para tu juego)

```cpp
class CommandRouter {
public:
    std::string handle(const std::string& input) {
        if (input == "hello") return "Hello, traveler.";
        if (input == "open inventory") return "Opening inventory.";
        return "I don't understand.";
    }
};
```
